# 手写单链表

## 函数声明

```C
// 1. 创建节点
Node* createNode(int val);

// 2. 头插
void pushFront(Node** head, int val);

// 3. 在指定节点之后插入新节点
void insertAfterNode(Node* prev, int val);

// 4. 尾插
void pushBack(Node** head, int val);

// 5. 按值删除第一个匹配的节点
void deleteByValue(Node** head, int val);

// 6. 按位置删除（0-based）
void deleteByPos(Node** head, int pos);

// 7. 按值查找第一个值为 val 的节点，返回节点指针
Node* findByValue(Node* head, int val);

// 8. 按下标查找节点（0-based），返回节点指针
Node* getByPos(Node* head, int pos);

// 9. 反转链表（迭代法，三指针）
void reverse(Node** head);

// 10. 打印链表
void printList(Node* head);

// 11. 销毁整个链表（释放所有节点）
void destroyList(Node** head);
```

## 实现代码

```C
#include <stdio.h>
#include <stdlib.h>
#include <stdbool.h>

typedef struct Node {
    int data;
    struct Node* next;
} Node;

// 1. 创建新节点
Node* createNode(int val) {
    Node* newNode = (Node*)malloc(sizeof(Node));
    if (newNode == NULL) { printf("内存分配失败\n"); exit(1); }
    newNode->data = val;
    newNode->next = NULL;
    return newNode;
}

// 2. 头插
void pushFront(Node** head, int val) {
    Node* newNode = createNode(val);
    newNode->next = *head;
    *head = newNode;
}

// 3. 在不带头结点的链表第 i 个位置插入元素 x
bool ListInsert(Node** head, int i, int x) {
    if (!head || i < 0) return false;
    Node* newNode = createNode(x);
    if (i == 0) {
        newNode->next = *head;
        *head = newNode;
        return true;
    }
    Node* prev = *head;
    int j = 0;
    while (j < i - 1 && prev) { prev = prev->next; j++; }
    if (!prev) { free(newNode); return false; }
    newNode->next = prev->next;
    prev->next = newNode;
    return true;
}

// 4. 尾插
void pushBack(Node** head, int val) {
    Node* newNode = createNode(val);
    if (*head == NULL) { *head = newNode; return; }
    Node* cur = *head;
    while (cur->next != NULL) cur = cur->next;
    cur->next = newNode;
}

// 5. 按值删除
void deleteByValue(Node** head, int val) {
    if (*head == NULL) return;
    if ((*head)->data == val) {
        Node* temp = *head;
        *head = (*head)->next;
        free(temp);
        return;
    }
    Node* cur = *head;
    while (cur->next != NULL && cur->next->data != val) cur = cur->next;
    if (cur->next == NULL) return;
    Node* del = cur->next;
    cur->next = del->next;
    free(del);
}

// 6. 按位置删除
void deleteByPos(Node** head, int pos) {
    if (*head == NULL || pos < 0) return;
    if (pos == 0) {
        Node* temp = *head;
        *head = (*head)->next;
        free(temp);
        return;
    }
    Node* cur = *head;
    int i = 0;
    while (cur->next != NULL && i < pos - 1) { cur = cur->next; i++; }
    if (cur->next == NULL) return;
    Node* del = cur->next;
    cur->next = del->next;
    free(del);
}

// 7. 按值查找
Node* findByValue(Node* head, int val) {
    Node* cur = head;
    while (cur != NULL) {
        if (cur->data == val) return cur;
        cur = cur->next;
    }
    return NULL;
}

// 8. 按下标查找
Node* getByPos(Node* head, int pos) {
    if (pos < 0 || head == NULL) return NULL;
    Node* cur = head;
    int i = 0;
    while (cur != NULL && i < pos) { cur = cur->next; i++; }
    return cur;
}

// 9. 反转链表
void reverse(Node** head) {
    Node* prev = NULL;
    Node* cur = *head;
    Node* next = NULL;
    while (cur != NULL) {
        next = cur->next;
        cur->next = prev;
        prev = cur;
        cur = next;
    }
    *head = prev;
}

// 10. 打印链表
void printList(Node* head) {
    Node* cur = head;
    while (cur != NULL) { printf("%d -> ", cur->data); cur = cur->next; }
    printf("NULL\n");
}

// 11. 销毁链表
void destroyList(Node** head) {
    Node* cur = *head;
    while (cur != NULL) {
        Node* temp = cur;
        cur = cur->next;
        free(temp);
    }
    *head = NULL;
}
```

# 带头节点的链表

## 函数声明

```C
Node* initList();

void pushBack(Node* head, int val);
void deleteByValue(Node* head, int val);

void reverse(Node* head);
void printList(Node* head);
void destroyList(Node** head);
```

## 实现代码

```C
Node* initList() { return createNode(-1); }

void pushFront(Node* head, int val) {
    Node* newNode = createNode(val);
    newNode->next = head->next;
    head->next = newNode;
}

bool ListInsert(Node* head, int i, int x) {
    if (!head || i < 0) return false;
    Node* prev = head;
    int j = -1;
    while (j < i - 1 && prev) { prev = prev->next; j++; }
    if (!prev) return false;
    Node* newNode = createNode(x);
    newNode->next = prev->next;
    prev->next = newNode;
    return true;
}

void pushBack(Node* head, int val) {
    Node* cur = head;
    while (cur->next != NULL) cur = cur->next;
    cur->next = createNode(val);
}

void deleteByValue(Node* head, int val) {
    Node* cur = head;
    while (cur->next != NULL && cur->next->data != val) cur = cur->next;
    if (cur->next == NULL) return;
    Node* del = cur->next;
    cur->next = del->next;
    free(del);
}

void deleteByPos(Node* head, int pos) {
    if (head->next == NULL || pos < 0) return;
    Node* cur = head;
    int i = 0;
    while (cur->next != NULL && i < pos) { cur = cur->next; i++; }
    if (cur->next == NULL) return;
    Node* del = cur->next;
    cur->next = del->next;
    free(del);
}

Node* findByValue(Node* head, int val) {
    Node* cur = head->next;
    while (cur != NULL) {
        if (cur->data == val) return cur;
        cur = cur->next;
    }
    return NULL;
}

Node* getByPos(Node* head, int pos) {
    if (pos < 0 || head->next == NULL) return NULL;
    Node* cur = head->next;
    int i = 0;
    while (cur != NULL && i < pos) { cur = cur->next; i++; }
    return cur;
}

void reverse(Node* head) {
    Node* prev = NULL;
    Node* cur = head->next;
    Node* next = NULL;
    while (cur != NULL) {
        next = cur->next;
        cur->next = prev;
        prev = cur;
        cur = next;
    }
    head->next = prev;
}

void printList(Node* head) {
    Node* cur = head->next;
    while (cur != NULL) { printf("%d -> ", cur->data); cur = cur->next; }
    printf("NULL\n");
}

void destroyList(Node** head) {
    Node* cur = *head;
    while (cur != NULL) {
        Node* temp = cur; cur = cur->next; free(temp);
    }
    *head = NULL;
}
```

# 有头节点 vs 无头节点 对比

|方面|不带头结点|带头结点|
|-|-|-|
|**初始化**|`Node* head = NULL;`|`Node* head = initList();`|
|**空表判断**|`head == NULL`|`head->next == NULL`|
|**头插/头删**|需修改头指针 `Node** head`|头指针固定，用 `Node* head`|
|**插入/删除首元**|需单独处理 `i==0`|与操作其他节点一致|
|**遍历起始**|`cur = head`|`cur = head->next`|
|**函数参数**|修改头指针用二级指针|除销毁外都用一级指针|

# 第 i 个位置插入题型总结

## 带头结点（推荐统一写法，j 从 -1 开始）

```C
bool ListInsert(Node* head, int i, int x) {
    if (!head || i < 0) return false;
    Node* prev = head;
    int j = -1;
    while (j < i - 1 && prev) { prev = prev->next; j++; }
    if (!prev) return false;
    Node* newNode = createNode(x);
    newNode->next = prev->next;
    prev->next = newNode;
    return true;
}
```

|i|最终 prev 指向|
|-|-|
|0|头结点（索引-1）✅ 头插无需特判|
|1|第0个数据结点|

## 不带头结点（必须分情况）

```C
void ListInsert(Node** head, int i, int x) {
    if (!head || i < 0) return;
    Node* newNode = createNode(x);
    if (i == 0) {
        newNode->next = *head;
        *head = newNode;
        return;
    }
    Node* prev = *head;
    int j = 0;
    while (j < i - 1 && prev) { prev = prev->next; j++; }
    if (!prev) { free(newNode); return; }
    newNode->next = prev->next;
    prev->next = newNode;
}
```

# *head 与 **head 的区别

- **`*head`**：一级指针，只能访问节点，不能修改外部头指针
- **`**head`**：二级指针，可以修改外部头指针本身

需要修改头指针的操作必须用二级指针：头插（i=0）、删除头节点、置空链表。

C++ 替代方案：用引用 `LNode* &head`。
