# 一.初始化方式不同的完整对比表（top = -1 vs top = 0）

|操作|`top = -1` 模式|`top = 0` 模式|
|-|-|-|
|**初始化**|`pst->top = -1;`|`pst->top = 0;`|
|**判空**|`return pst->top == -1;`|`return pst->top == 0;`|
|判满|`top + 1 == capacity`|`top == capacity`|
|**压栈**|`pst->a[++pst->top] = x;`|`pst->a[pst->top++] = x;`|
|**出栈**|`pst->top--;`|`pst->top--;`（相同）|
|**取栈顶**|`return pst->a[pst->top];`|`return pst->a[pst->top - 1];`|
|**元素个数**|`pst->top + 1`|`pst->top`|

# 二.建立一套完整的顺序栈

## 头文件 Stack.h

```C
#ifndef STACK_H
#define STACK_H

#include <stdbool.h>
#include <stddef.h>

typedef int STDataType;

typedef struct Stack {
    STDataType* a;
    int top;
    int capacity;
} ST;

void STInit(ST* pst);
void STDestroy(ST* pst);
void STPush(ST* pst, STDataType x);
void STPop(ST* pst);
STDataType STTop(ST* pst);
bool STEmpty(ST* pst);
int STSize(ST* pst);

#endif
```

## 源文件 Stack.c

```C
#include "Stack.h"
#include <stdio.h>
#include <stdlib.h>
#include <assert.h>

void STInit(ST* pst) {
    assert(pst);
    pst->a = NULL;
    pst->top = 0;
    pst->capacity = 0;
}

void STDestroy(ST* pst) {
    assert(pst);
    free(pst->a);
    pst->a = NULL;
    pst->top = 0;
    pst->capacity = 0;
}

static void STCheckCapacity(ST* pst) {
    if (pst->top == pst->capacity) {
        int newCapacity = pst->capacity == 0 ? 4 : pst->capacity * 2;
        STDataType* tmp = (STDataType*)realloc(pst->a, newCapacity * sizeof(STDataType));
        if (tmp == NULL) { perror("realloc fail"); exit(-1); }
        pst->a = tmp;
        pst->capacity = newCapacity;
    }
}

void STPush(ST* pst, STDataType x) {
    assert(pst);
    STCheckCapacity(pst);
    pst->a[pst->top++] = x;
}

void STPop(ST* pst) {
    assert(pst);
    assert(!STEmpty(pst));
    pst->top--;
}

STDataType STTop(ST* pst) {
    assert(pst);
    assert(!STEmpty(pst));
    return pst->a[pst->top - 1];
}

bool STEmpty(ST* pst) {
    assert(pst);
    return pst->top == 0;
}

int STSize(ST* pst) {
    assert(pst);
    return pst->top;
}
```

|操作|代码|解释|
|-|-|-|
|初始化|`top = 0; a = NULL; capacity = 0`|空栈|
|压栈|`a[top++] = x`|先放元素，再移动指针|
|取栈顶|`a[top-1]`|因为 top 指向下一个空位|
|判空|`top == 0`|没有元素|
|元素个数|`top`|直接等于 top|

# 三.队列中头删尾插操作

## 结构体定义

```C
typedef struct QueueNode {
    struct QueueNode* next;
    QDataType val;
} QNode;

typedef struct Queue {
    QNode* phead;
    QNode* ptail;
} Queue;
```

## 封装版本 vs 未封装版本

|方面|封装版本|未封装版本|
|-|-|-|
|参数数量|1 个（Queue*）|2 个（QNode**）|
|语义清晰|队列是一个整体|头尾分开，逻辑分散|
|不易出错|不会传错顺序|容易搞反顺序|

**封装版本**：通过 `pq->phead` 和 `pq->ptail` 直接修改，代码清晰。
**未封装版本**：需要 `*pphead` 和 `*pptail` 解引用两次。

```C
Queue q;
QueuePush(&q, 1);   // 封装版本
```

# 四.完整的队列实现

```C
#include <stdio.h>
#include <stdlib.h>
#include <stdbool.h>
#include <assert.h>

typedef int QDataType;

typedef struct QueueNode {
    QDataType val;
    struct QueueNode* next;
} QNode;

typedef struct Queue {
    QNode* phead;
    QNode* ptail;
    int size;
} Queue;

void QueueInit(Queue* pq) {
    assert(pq);
    pq->phead = NULL;
    pq->ptail = NULL;
    pq->size = 0;
}

void QueueDestroy(Queue* pq) {
    assert(pq);
    QNode* cur = pq->phead;
    while (cur) {
        QNode* next = cur->next;
        free(cur);
        cur = next;
    }
    pq->phead = pq->ptail = NULL;
    pq->size = 0;
}

void QueuePush(Queue* pq, QDataType x) {
    assert(pq);
    QNode* newNode = (QNode*)malloc(sizeof(QNode));
    if (newNode == NULL) { perror("malloc failed"); return; }
    newNode->val = x;
    newNode->next = NULL;

    if (pq->ptail == NULL) {
        pq->phead = pq->ptail = newNode;
    } else {
        pq->ptail->next = newNode;
        pq->ptail = newNode;
    }
    pq->size++;
}

void QueuePop(Queue* pq) {
    assert(pq);
    assert(pq->phead);
    QNode* del = pq->phead;
    pq->phead = pq->phead->next;
    free(del);
    if (pq->phead == NULL) pq->ptail = NULL;
    pq->size--;
}

QDataType QueueFront(Queue* pq) {
    assert(pq);
    assert(pq->size != 0);
    return pq->phead->val;
}

QDataType QueueBack(Queue* pq) {
    assert(pq);
    assert(pq->ptail);
    return pq->ptail->val;
}

bool QueueEmpty(Queue* pq) {
    assert(pq);
    return pq->phead == NULL;
}

int QueueSize(Queue* pq) {
    assert(pq);
    return pq->size;
}
```
